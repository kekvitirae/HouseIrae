# Regular Holy War CB

drow_holy_war = {
	name = CB_NAME_RELIGIOUS
	war_name = WAR_NAME_RELIGIOUS
	sprite = 9
	truce_days = 3650
	is_permanent = yes
	is_holy_war = yes
	check_de_jure_tier = DUKE # this scans all dejure duchies for the counties which are held by or vassals(or below) of selected character. Only valid if is_permanent = yes

	allowed_to_target_tributaries = no
	
	can_use = {
		is_proper_alternate_start = yes
		ROOT = {
			NOT = { has_character_flag = npc_laela_irae }
			is_nomadic = no
			NOT = { same_realm = FROM }
			OR = {
				AND = {
					religion = eilistraee_religion
					FROM = { 
						OR = {
							religion = lolth_religion
							religion = vhaeraun_religion
						}
					}
				}
				AND = {
					religion = lolth_religion
					FROM = { 
						OR = {
							religion = eilistraee_religion
							religion = vhaeraun_religion
						}
					}
				}
				AND = {
					religion = vhaeraun_religion
					FROM = { 
						OR = {
							religion = eilistraee_religion
							religion = lolth_religion
						}
					}
				}
			}
			mercenary = no
		}
	}
	
	can_use_title = {
		# The attacker needs at least one county in the target kingdom, or a border, or be only two sea zones away from one of your counties
		any_direct_de_jure_vassal_title = {
			OR = {
				holder_scope = {
					OR = {
						character = ROOT
						is_liege_or_above = ROOT
					}
				}
				location = {
					any_neighbor_province = {
						OR = {
							AND = {
								has_owner = yes
								owner = {
									OR = {
										character = ROOT
										is_liege_or_above = ROOT
									}
								}
							}
							AND = {
								is_land = no
								any_neighbor_province = {
									OR = {
										AND = {
											has_owner = yes
											owner = {
												OR = {
													character = ROOT
													is_liege_or_above = ROOT
												}
											}
										}
										AND = {
											is_land = no
											any_neighbor_province = {
												owner = {
													OR = {
														character = ROOT
														is_liege_or_above = ROOT
													}
												}
											}
										}
									}
								}
							}
						}
					}
				}
			}
		}
	}

	is_valid = {
		ROOT = {
			NOT = { same_realm = FROM }
			OR = {
				AND = {
					religion = eilistraee_religion
					FROM = { 
						OR = {
							religion = lolth_religion
							religion = vhaeraun_religion
						}
					}
				}
				AND = {
					religion = lolth_religion
					FROM = { 
						OR = {
							religion = eilistraee_religion
							religion = vhaeraun_religion
						}
					}
				}
				AND = {
					religion = vhaeraun_religion
					FROM = { 
						OR = {
							religion = eilistraee_religion
							religion = lolth_religion
						}
					}
				}
			}
		}
	}
	
	on_success = {
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = { 
				participation_scaled_piety = 100
				participation_scaled_prestige = 50
				if = {
					limit = {
						uses_decadence = yes
					}
					participation_scaled_decadence = -10
				}
				hidden_tooltip = {
					if = {
						limit = {
							has_ambition = obj_strengthen_religion
						}
						change_variable = { which = strengthen_religion value = 1 }
					}
				}
			}
		}
		
		ROOT = { 
			participation_scaled_piety = 100
			participation_scaled_prestige = 50
			if = {
				limit = {
					uses_decadence = yes
				}
				participation_scaled_decadence = -10
			}
			religion_authority = {
				modifier = won_holy_war
				years = 20
			}
			hidden_tooltip = {
				if = {
					limit = {
						has_ambition = obj_strengthen_religion
					}
					change_variable = { which = strengthen_religion value = 1 }
				}
			}
		}
		FROM = {
			if = {
				limit = {
					NOT = {
						has_character_modifier = known_satanist
					}
				}
				religion_authority = {
					modifier = lost_holy_war
					years = 20
				}		
			}
		}
		FROM = { piety = -100 }
		FROM = { prestige = -50 }
	}

	on_success_title = {
		if = { # Standard Holy War behaviour for Non-Muslims and Muslims fighting other Muslims
			limit = {
				OR = {
					ROOT = {
						NOT = {
							religion_group = muslim
						}
					}
					AND = {
						ROOT = {
							religion_group = muslim
						}
						FROM = {
							religion_group = muslim
						}
					}
				}
			}
			custom_tooltip = {
				text = religious_cb_succ_tip
				hidden_tooltip = {
					ROOT = {
						vassalize_or_take_under_title = {
							title = PREV
							enemy = FROM
							same_religion = yes # Only vassalize rulers of my religion
							is_religious = yes
							type = holy_war
						}
					}
				}
			}
		}
		if = { # Muslims fighting Non-Muslims preserve vassals
			limit = {
				ROOT = {
					religion_group = muslim
				}
				FROM = {
					NOT = {
						religion_group = muslim
					}
				}
			}
			custom_tooltip = {
				text = religious_cb_succ_muslim_tip
				hidden_tooltip = {
					ROOT = {
						vassalize_or_take_under_title = {
							title = PREV
							enemy = FROM
							is_religious = yes
							type = holy_war
						}
					}
				}
			}
		}
	}

	on_fail = {
		ROOT = { 
			piety = -50 
			prestige = -100
		}
	}

	on_reverse_demand = {
		ROOT = {
			transfer_scaled_wealth = {
				to = FROM
				value = 4.0
			}
			piety = -75
			prestige = -150
		}
		FROM = {
			participation_scaled_piety = 200
			participation_scaled_prestige = 100
			if = {
				limit = {
					uses_decadence = yes
				}
				participation_scaled_decadence = -10
			}
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { 
				participation_scaled_piety = 200
				participation_scaled_prestige = 100
				if = {
					limit = {
						uses_decadence = yes
					}
					participation_scaled_decadence = -10
				}
				hidden_tooltip = {
					if = {
						limit = {
							has_ambition = obj_strengthen_religion
						}
						change_variable = { which = strengthen_religion value = 1 }
					}
				}
			}
		}
		any_attacker = {
			limit = {
				character = ROOT
				uses_decadence = yes
			}
			participation_scaled_decadence = 20
		}
		any_attacker = {
			limit = {
				NOT = { character = ROOT }
				uses_decadence = yes
			}
			hidden_tooltip = { 
				participation_scaled_decadence = 20
			}
		}
		ROOT = { 
			religion_authority = {
				modifier = lost_holy_war
				years = 20
			}
		}
		FROM = { 
			religion_authority = {
				modifier = won_holy_war
				years = 20
			}
			hidden_tooltip = {
				if = {
					limit = {
						has_ambition = obj_strengthen_religion
					}
					change_variable = { which = strengthen_religion value = 1 }
				}
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
	
	ai_will_do = { # Laela will only use her unique CB
		factor = 1
		modifier = {
			factor = 0
			ROOT = {
				has_character_flag = npc_laela_irae
			}
		}
	}
}

	allowed_to_target_tributaries = no
	
	can_use_title = {
		ROOT = {
			has_character_flag = npc_laela_irae # Only Laela can do this...
			primary_title = { title = k_house_laela } # ...and only if she has control of her kingdom
			ai = yes
			is_nomadic = no
		}
		
		ROOT = {
			NOT = { same_realm = FROM }
		}
		
		FROM = {
			NOT = { # Laela has no interest in crap lands.
				is_nomadic = yes
			}
			NAND = { # Laela will not attack fellow dark elves... unless she hates them.
				trait = houseirae_drow
				OR = {
					is_female = no
					NOT = { religion = lolth_religion }
				}
			}
			any_realm_province = { # Laela will not attack any provinces that are not adjacent to her, nor attack any province outside of her ambitions
				OR = {
					has_global_flag = scenario_initialized # After The End compatibility. 
					has_global_flag = scenario_initialised # AGoT compatibility.
					# BEWARE: She will gank everyone without a region set
					OR = {
						region = world_europe_north
						region = world_europe_west_brittania
						region = custom_russia
					}
				}
				OR = {
					ROOT = {
						any_realm_province = {
							port = yes
							can_naval_path_to = PREVPREV
						}
					}
					any_neighbor_province = {
						owner = {
							OR = {
								character = ROOT

			hidden_tooltip = {
				if = {
					limit = {

					}
					change_variable = { which = strengthen_religion value = 1 }
				}
			}
		}
		FROM = { piety = -100 }
		FROM = { prestige = -50 }
	}

	on_success_title = {
		custom_tooltip = {
			text = religious_cb_succ_tip
			hidden_tooltip = {
				ROOT = {
					occupy_minors_of_occupied_settlements = FROM
					gain_all_occupied_titles = { who = FROM type = invasion }
					laela_converts_tribals = yes
				}
			}
		}
		ROOT = {
			character_event = {
				id = Laela.22
			}
		}
	}

	on_fail = {
		ROOT = { 
			piety = -50 
			prestige = -100
		}

	on_reverse_demand = {

		}
		any_defender = {
			limit = { NOT = { character = FROM } }

			mercenary = no
			is_nomadic = no
			trait = houseirae_laelas_consort
		}
	}

	can_use_title = {
		tier = count
		OR = {
			FROM = {
				has_landed_title = PREV
			}
			holder_scope = {
				is_liege_or_above = FROM
				NOT = { same_realm = ROOT }
			}
		}

		NOT = {
			OR = {
				ROOT = {
					has_landed_title = PREV
				}

					is_liege_or_above = ROOT
				}
			}
		}
		
		location = {

				owner = {
					OR = {

						is_liege_or_above = ROOT
					}
				}
			}
		}
	}
	
	is_valid_title = {
		OR = {
			ROOT = {
				is_nomadic = yes
			}
			ROOT = {
				NOT = { same_realm = FROM }
			}
		}
	
		OR = {
			FROM = {
				has_landed_title = PREV
			}
			holder_scope = {
				is_liege_or_above = FROM
			}
		}
	}

	on_success_title = {
		usurp_title_plus_barony_if_unlanded = { target = ROOT type = invasion }
		
		any_de_jure_vassal_title = { # take all baronies too
			limit = {
				has_holder = yes
				holder_scope = {
					OR = {
						character = FROM
						is_liege_or_above = FROM
						AND = {
							is_liege_or_above = ROOT
							NOT = { culture = ROOT }
							NOT = { religion = ROOT }
						}
					}
				}
			}

			usurp_title = ROOT
		}
		
		ROOT = {
			participation_scaled_prestige = 100
			participation_scaled_piety = 50
			narrative_event = {
				id = Laela.221
			}
		}
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = { 
				participation_scaled_prestige = 100 
				participation_scaled_piety = 50
			}
		}
		
		if = {
			limit = {
				NOT = { FROM = { religion = ROOT } }
			}
			ROOT = {
				religion_authority = {
					modifier = won_county_conquest
				}
			}
		}
	}

	on_fail_title = {
		ROOT = {
			prestige = -100
		}
		FROM = {
			participation_scaled_prestige = 50
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { participation_scaled_prestige = 50 }
		}
	}

	on_reverse_demand = {
		ROOT = {
			prestige = -200
			transfer_scaled_wealth = {
				to = FROM
				value = 4.0
			}
		}
		FROM = {
			participation_scaled_prestige = 100
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { participation_scaled_prestige = 100 }
		}
